## Modules

Import
```javascript
import {name} from "<path>";
```
or with `alias`
```javascript
import {name as n} from "<path>";

n('John','Doe')
```


Export
```javascript

export function name(first, second) {
  console.log(`${first} ${second}`)
}
name('John','Doe') // John Doe
```
or
```javascript

function name(first, second) {
  console.log(`${first} ${second}`)
}
name('John','Doe') // John Doe

export { name };
```
or
```javascript

export function name(first, second) {
  console.log(`${first} ${second}`)
}
name('John','Doe') // John DOe
```
